CREATE TABLE PRODUCTS 
(
PROD_ID NUMERIC (6) NOT NULL,
PROD_CAT_ID NUMERIC (5) NOT NULL,
PROD_NAME VARCHAR (50) NOT NULL UNIQUE,
PROD_SIZE VARCHAR(3) NOT NULL,
PROD_QUANTITY NUMERIC (3) NOT NULL,
PROD_PRICE DECIMAL (3,2) NOT NULL,
CONSTRAINT PK_PROD_ID PRIMARY KEY (PROD_ID),
CONSTRAINT FK_PROD_CAT_ID FOREIGN KEY (PROD_CAT_ID) REFERENCES CATEGORIES_PRODUCT(PROD_CAT_ID));
